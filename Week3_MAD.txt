package com.example.basicuserinfo;

import android.os.Bundle;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.TextView;
import android.widget.Toast;
import android.widget.ImageView;
import android.view.View;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText etName;
    RadioGroup rgGender;
    RadioButton rbMale, rbFemale;
    CheckBox cbReading, cbSinging, cbCricket;
    Button btnSubmit, btnReset;
    TextView tvResult;
    ImageView imageProfile;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.editTextName);
        rgGender = findViewById(R.id.radioGroupGender);
        rbMale = findViewById(R.id.radioButtonMale);
        rbFemale = findViewById(R.id.radioButtonFemale);
        cbReading = findViewById(R.id.checkBoxReading);
        cbSinging = findViewById(R.id.checkBoxSinging);
        cbCricket = findViewById(R.id.checkBoxCricket);
        btnSubmit = findViewById(R.id.buttonSubmit);
        btnReset = findViewById(R.id.buttonReset);
        tvResult = findViewById(R.id.textViewDisplayInfo);
        imageProfile = findViewById(R.id.imageViewProfile);

        btnSubmit.setOnClickListener(v -> handleSubmit());
        btnReset.setOnClickListener(v -> handleReset());

        rgGender.setOnCheckedChangeListener((group, checkedId) -> {
            if (checkedId == R.id.radioButtonMale) {
                Toast.makeText(this, "Gender: Male selected", Toast.LENGTH_SHORT).show();
            } else if (checkedId == R.id.radioButtonFemale) {
                Toast.makeText(this, "Gender: Female selected", Toast.LENGTH_SHORT).show();
            }
        });
    }

    private void handleSubmit() {
        String name = etName.getText().toString().trim();
        if (name.isEmpty()) {
            Toast.makeText(this, "Please enter your name", Toast.LENGTH_SHORT).show();
            return;
        }

        int selectedGenderId = rgGender.getCheckedRadioButtonId();
        if (selectedGenderId == -1) {
            Toast.makeText(this, "Please select a gender", Toast.LENGTH_SHORT).show();
            return;
        }

        RadioButton selectedRadioButton = findViewById(selectedGenderId);
        String gender = selectedRadioButton.getText().toString();

        StringBuilder hobbiesBuilder = new StringBuilder();
        if (cbReading.isChecked()) hobbiesBuilder.append(cbReading.getText()).append(", ");
        if (cbSinging.isChecked()) hobbiesBuilder.append(cbSinging.getText()).append(", ");
        if (cbCricket.isChecked()) hobbiesBuilder.append(cbCricket.getText()).append(", ");

        String finalMessage;
        if (hobbiesBuilder.length() > 0) {
            String hobbies = hobbiesBuilder.substring(0, hobbiesBuilder.length() - 2);
            finalMessage = "Hello " + name + ", you are " + gender + " and your hobbies are " + hobbies + ".";
        } else {
            finalMessage = "Hello " + name + ", you are " + gender + ".";
        }

        tvResult.setText("Hello " + name);
        Toast.makeText(this, finalMessage, Toast.LENGTH_LONG).show();
        imageProfile.setVisibility(View.VISIBLE);
    }

    private void handleReset() {
        etName.setText("");
        rgGender.clearCheck();
        cbReading.setChecked(false);
        cbSinging.setChecked(false);
        cbCricket.setChecked(false);
        tvResult.setText("");
        imageProfile.setVisibility(View.GONE);
    }
}
